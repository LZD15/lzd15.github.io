import pygame import math import random # 初始化 pygame.init() screen = pygame.display.set_mode((800, 600)) pygame.display.set_caption("Fireworks") clock = pygame.time.Clock() # 定义烟花颜色 colors = [ (255, 0, 0), # Red (255, 165, 0), # Orange (255, 255, 0), # Yellow (0, 255, 0), # Green (0, 0, 255), # Blue (75, 0, 130), # Indigo (255, 0, 255), # Pink (128, 0, 128), # Purple ] # 定义烟花粒子 class Firework: def __init__(self): self.x = random.randint(0, 800) self.y = 600 self.size = 20 self.color = random.choice(colors) self.speed = random.uniform(1, 5) self.direction = random.uniform(0, 2 * math.pi) def move(self): self.y -= self.speed * math.sin(self.direction) self.x += self.speed * math.cos(self.direction) def draw(self): pygame.draw.circle(screen, self.color, (int(self.x), int(self.y)), self.size) # 创建烟花粒子 fireworks = [Firework() for _ in range(100)] # 游戏循环 running = True while running: screen.fill((0, 0, 0)) # 绘制烟花 for firework in fireworks: firework.move() firework.draw() # 检测关闭窗口 for event in pygame.event.get(): if event.type == pygame.QUIT: running = False # 更新屏幕 pygame.display.flip() clock.tick(60) # 退出 pygame.quit()
